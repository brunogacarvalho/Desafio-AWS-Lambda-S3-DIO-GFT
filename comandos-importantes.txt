# Comandos Importantes - AWS Lambda + S3 + DynamoDB com LocalStack

## LocalStack

# Verificar se o LocalStack está rodando
curl http://localhost:4566/_localstack/health

## Amazon S3

# Criar bucket
awslocal s3 mb s3://nome-do-bucket

# Listar buckets
awslocal s3 ls

# Fazer upload de arquivo para o bucket
awslocal s3 cp arquivo.json s3://nome-do-bucket/

# Listar arquivos dentro do bucket
awslocal s3 ls s3://nome-do-bucket/

## Amazon DynamoDB

# Criar tabela
awslocal dynamodb create-table \
  --table-name nome-da-tabela \
  --attribute-definitions AttributeName=id,AttributeType=S \
  --key-schema AttributeName=id,KeyType=HASH \
  --billing-mode PAY_PER_REQUEST

# Listar tabelas
awslocal dynamodb list-tables

# Escanear (ver dados) de uma tabela
awslocal dynamodb scan --table-name nome-da-tabela

## AWS Lambda

# Criar função Lambda
awslocal lambda create-function \
  --function-name nome-da-funcao \
  --runtime python3.9 \
  --role arn:aws:iam::000000000000:role/lambda-role \
  --handler grava_db.lambda_handler \
  --zip-file fileb://lambda_function.zip

# Listar funções Lambda
awslocal lambda list-functions

# Adicionar permissão para o S3 acionar a Lambda
awslocal lambda add-permission \
  --function-name nome-da-funcao \
  --statement-id s3-trigger \
  --action lambda:InvokeFunction \
  --principal s3.amazonaws.com \
  --source-arn arn:aws:s3:::nome-do-bucket

# Configurar trigger do S3 para acionar a Lambda
awslocal s3api put-bucket-notification-configuration \
  --bucket nome-do-bucket \
  --notification-configuration file://notification_roles.json
